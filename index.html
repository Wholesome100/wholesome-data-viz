# Calculate average cost of living by country and show the top 10
top_10_countries = (
    df_cola.group_by("Country")
    .agg(pl.mean("Cost of Living Index").alias("Avg Cost of Living"))
    .sort("Avg Cost of Living", descending=True)
    .head(10)
)
